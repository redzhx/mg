    class RandomImg extends React.Component {
     constructor() {
     super();
     this.state = {
     bgImageUrl: require('./default.jpeg')
     }
     }
     
     componentDidMount() {
     this.generateImage();
     }
     
     generateImage(){
     fetch(`https://source.unsplash.com/random/?people`).then((response) => {
     preloadImage(response.url);
     })
    	}
     
     preloadImage(url) {
     let img = new Image();
     img.src = url;
     img.onload = () => {
     this.setState({
     bgImageUrl: url
     });
     }
    	}
     
     render() {
     return <img src={bgImageUrl} />
     }
    }
